Application code:
package com.objectplanet.chart.testing;

import com.objectplanet.chart.*;
import java.awt.*;
public class Bar {

     public static void main(String[] argv) {
         double[] values = new double[] {10, 20, 30, 40, 50};
         Color[] c = new Color[] {new Color(0x0587D9)};
         BarChart chart = new BarChart();

         chart.setSampleCount(5);
         chart.setSampleValues(0, values);
         chart.setSampleColors(c);
         chart.setBackground(Color.white);
         chart.set3DModeOn(true);
         chart.set3DDepth(5);
         chart.setBarWidth(0.6);
         chart.setBarOutlineOn(false);
         chart.setRangeLabelsOn(0, false);
         chart.setRange(0, 50);

         Frame f = new Frame();
         f.setSize(300,220);
         f.add("Center", chart);
         f.show();
     }
}
